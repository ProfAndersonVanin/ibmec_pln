5. SENTENCE TRANSFORMERS + FAISS — REFERÊNCIA RAG

  from sentence_transformers import SentenceTransformer
  import faiss, numpy as np

  # Carregar modelo de embedding
  modelo = SentenceTransformer('all-MiniLM-L6-v2')

  # Gerar embeddings
  embeddings = modelo.encode(textos, convert_to_numpy=True)

  # Criar índice FAISS
  dim = embeddings.shape[1]
  embeddings_norm = embeddings.astype('float32')
  faiss.normalize_L2(embeddings_norm)

  indice = faiss.IndexFlatIP(dim)  # Inner Product (cosseno)
  indice.add(embeddings_norm)

  # Buscar
  def buscar(pergunta, top_k=3):
      emb = modelo.encode([pergunta], convert_to_numpy=True).astype('float32')
      faiss.normalize_L2(emb)
      scores, indices = indice.search(emb, top_k)
      return [(textos[i], scores[0][j]) for j, i in enumerate(indices[0])]

  resultados = buscar("O que é LSTM?", top_k=3)
